-- 1. Tabla para guardar resultados de tests (si no la tienes creada aún)
CREATE TABLE TEST_RESULTADOS (
  id_test NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
  nombre_test VARCHAR2(100),
  resultado VARCHAR2(20),
  mensaje VARCHAR2(500),
  fecha DATE DEFAULT SYSDATE
);

-- 2. Procedimiento para insertar resultados
CREATE OR REPLACE PROCEDURE P_INSERTAR_RESULTADO_TEST(
  p_nombre_test IN VARCHAR2,
  p_resultado IN VARCHAR2,
  p_mensaje IN VARCHAR2
) AS
BEGIN
  INSERT INTO TEST_RESULTADOS(nombre_test, resultado, mensaje)
  VALUES (p_nombre_test, p_resultado, p_mensaje);
  COMMIT;
EXCEPTION
  WHEN OTHERS THEN
    DBMS_OUTPUT.PUT_LINE('Error insertando resultado de test: ' || SQLERRM);
END;
/

-- 3. Procedimiento para ejecutar todos los tests de F_LISTA_CATEGORIAS_PRODUCTO
CREATE OR REPLACE PROCEDURE P_RUN_TESTS_F_LISTA_CATEGORIAS_PRODUCTO AS
  v_result VARCHAR2(4000);
BEGIN

  -- Test 1: Producto válido con varias categorías
  BEGIN
    v_result := PKG_ADMIN_PRODUCTOS.F_LISTA_CATEGORIAS_PRODUCTO('1234567890123', 1001);
    -- Asumimos que debe devolver algo parecido a "Electrónica ; Móviles ; Regalo día de la Madre"
    IF INSTR(v_result, 'Electrónica') > 0 AND INSTR(v_result, 'Móviles') > 0 THEN
      P_INSERTAR_RESULTADO_TEST('F_LISTA_CATEGORIAS_PRODUCTO - producto válido con categorías', 'OK', 'Categorías obtenidas: ' || v_result);
    ELSE
      P_INSERTAR_RESULTADO_TEST('F_LISTA_CATEGORIAS_PRODUCTO - producto válido con categorías', 'FAIL', 'Categorías inesperadas: ' || v_result);
    END IF;
  EXCEPTION
    WHEN OTHERS THEN
      P_INSERTAR_RESULTADO_TEST('F_LISTA_CATEGORIAS_PRODUCTO - producto válido con categorías', 'FAIL', SQLERRM);
  END;

  -- Test 2: Producto válido sin categorías asignadas
  BEGIN
    v_result := PKG_ADMIN_PRODUCTOS.F_LISTA_CATEGORIAS_PRODUCTO('9876543210987', 1001);
    IF v_result = 'Sin categoría' OR v_result IS NULL OR TRIM(v_result) = '' THEN
      P_INSERTAR_RESULTADO_TEST('F_LISTA_CATEGORIAS_PRODUCTO - producto válido sin categorías', 'OK', 'Resultado esperado: ' || NVL(v_result, 'NULL'));
    ELSE
      P_INSERTAR_RESULTADO_TEST('F_LISTA_CATEGORIAS_PRODUCTO - producto válido sin categorías', 'FAIL', 'Resultado inesperado: ' || v_result);
    END IF;
  EXCEPTION
    WHEN OTHERS THEN
      P_INSERTAR_RESULTADO_TEST('F_LISTA_CATEGORIAS_PRODUCTO - producto válido sin categorías', 'FAIL', SQLERRM);
  END;

  -- Test 3: Producto inexistente (debe lanzar NO_DATA_FOUND)
  BEGIN
    v_result := PKG_ADMIN_PRODUCTOS.F_LISTA_CATEGORIAS_PRODUCTO('0000000000000', 1001);
    -- Si no lanza excepción, test falla
    P_INSERTAR_RESULTADO_TEST('F_LISTA_CATEGORIAS_PRODUCTO - producto inexistente', 'FAIL', 'No se lanzó NO_DATA_FOUND');
  EXCEPTION
    WHEN NO_DATA_FOUND THEN
      P_INSERTAR_RESULTADO_TEST('F_LISTA_CATEGORIAS_PRODUCTO - producto inexistente', 'OK', 'Excepción NO_DATA_FOUND capturada correctamente');
    WHEN OTHERS THEN
      P_INSERTAR_RESULTADO_TEST('F_LISTA_CATEGORIAS_PRODUCTO - producto inexistente', 'FAIL', SQLERRM);
  END;

  -- Test 4: Cuenta inválida para producto existente (debe lanzar NO_DATA_FOUND)
  BEGIN
    v_result := PKG_ADMIN_PRODUCTOS.F_LISTA_CATEGORIAS_PRODUCTO('1234567890123', 9999);
    P_INSERTAR_RESULTADO_TEST('F_LISTA_CATEGORIAS_PRODUCTO - cuenta inválida', 'FAIL', 'No se lanzó NO_DATA_FOUND');
  EXCEPTION
    WHEN NO_DATA_FOUND THEN
      P_INSERTAR_RESULTADO_TEST('F_LISTA_CATEGORIAS_PRODUCTO - cuenta inválida', 'OK', 'Excepción NO_DATA_FOUND capturada');
    WHEN OTHERS THEN
      P_INSERTAR_RESULTADO_TEST('F_LISTA_CATEGORIAS_PRODUCTO - cuenta inválida', 'FAIL', SQLERRM);
  END;

  -- Test 5: Usuario sin acceso a la cuenta (debe lanzar EXCEPTION_SIN_ACCESO_CUENTA)
  BEGIN
    v_result := PKG_ADMIN_PRODUCTOS.F_LISTA_CATEGORIAS_PRODUCTO('1234567890123', 2001);
    P_INSERTAR_RESULTADO_TEST('F_LISTA_CATEGORIAS_PRODUCTO - sin acceso a la cuenta', 'FAIL', 'No se lanzó EXCEPTION_SIN_ACCESO_CUENTA');
  EXCEPTION
    WHEN PKG_ADMIN_PRODUCTOS.EXCEPTION_SIN_ACCESO_CUENTA THEN
      P_INSERTAR_RESULTADO_TEST('F_LISTA_CATEGORIAS_PRODUCTO - sin acceso a la cuenta', 'OK', 'Excepción capturada correctamente');
    WHEN OTHERS THEN
      P_INSERTAR_RESULTADO_TEST('F_LISTA_CATEGORIAS_PRODUCTO - sin acceso a la cuenta', 'FAIL', SQLERRM);
  END;

  -- Test 6: Verificar que concatenación tiene separadores correctos
  BEGIN
    v_result := PKG_ADMIN_PRODUCTOS.F_LISTA_CATEGORIAS_PRODUCTO('1234567890123', 1001);
    IF INSTR(v_result, ' ; ') > 0 THEN
      P_INSERTAR_RESULTADO_TEST('F_LISTA_CATEGORIAS_PRODUCTO - formato concatenación categorías', 'OK', 'Separadores presentes en resultado: ' || v_result);
    ELSE
      P_INSERTAR_RESULTADO_TEST('F_LISTA_CATEGORIAS_PRODUCTO - formato concatenación categorías', 'FAIL', 'Separadores no encontrados: ' || v_result);
    END IF;
  EXCEPTION
    WHEN OTHERS THEN
      P_INSERTAR_RESULTADO_TEST('F_LISTA_CATEGORIAS_PRODUCTO - formato concatenación categorías', 'FAIL', SQLERRM);
  END;

END;
/

-- Para ejecutar y ver los resultados
BEGIN
    P_RUN_TESTS_F_LISTA_CATEGORIAS_PRODUCTO;
END;
/

SELECT * FROM TEST_RESULTADOS ORDER BY fecha DESC;
