-- FUNCIÓN MOCK
CREATE OR REPLACE FUNCTION F_OBTENER_PLAN_CUENTA_TEST(p_cuenta_id IN NUMBER) 
RETURN PLAN_TEST%ROWTYPE AS
    v_plan PLAN_TEST%ROWTYPE;
BEGIN
    SELECT * INTO v_plan FROM PLAN_TEST WHERE ID = p_cuenta_id;
    RETURN v_plan;
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        RAISE_APPLICATION_ERROR(-20001, 'Plan no encontrado para cuenta: ' || p_cuenta_id);
END;
/

-- FUNCIÓN DE PRUEBA
CREATE OR REPLACE FUNCTION F_VALIDAR_PLAN_SUFICIENTE_TEST(p_cuenta_id IN NUMBER) 
RETURN VARCHAR2 AS
    v_plan PLAN_TEST%ROWTYPE;
    v_count_productos NUMBER;
    v_count_activos NUMBER;
    v_count_cat_producto NUMBER;
    v_count_cat_activos NUMBER;
    v_count_relaciones NUMBER;
BEGIN
    v_plan := F_OBTENER_PLAN_CUENTA_TEST(p_cuenta_id);
    
    SELECT COUNT(*) INTO v_count_productos FROM PRODUCTO_TEST WHERE CUENTA_ID = p_cuenta_id;
    SELECT COUNT(*) INTO v_count_activos FROM ACTIVO_TEST WHERE CUENTA_ID = p_cuenta_id;
    SELECT COUNT(*) INTO v_count_cat_producto FROM CATEGORIA_TEST WHERE CUENTA_ID = p_cuenta_id;
    SELECT COUNT(*) INTO v_count_cat_activos FROM CATEGORIA_ACTIVO_TEST WHERE CUENTA_ID = p_cuenta_id;
    SELECT COUNT(*) INTO v_count_relaciones 
    FROM RELACIONADO_TEST r
    JOIN PRODUCTO_TEST p ON r.PRODUCTO_GTIN = p.GTIN AND r.PRODUCTO_CUENTA_ID = p.CUENTA_ID
    WHERE p.CUENTA_ID = p_cuenta_id;
    
    IF v_count_productos > v_plan.PRODUCTOS THEN
        RETURN 'INSUFICIENTE: PRODUCTOS';
    ELSIF v_count_activos > v_plan.ACTIVOS THEN
        RETURN 'INSUFICIENTE: ACTIVOS';
    ELSIF v_count_cat_producto > v_plan.CATEGORIASPRODUCTO THEN
        RETURN 'INSUFICIENTE: CATEGORIASPRODUCTO';
    ELSIF v_count_cat_activos > v_plan.CATEGORIASACTIVOS THEN
        RETURN 'INSUFICIENTE: CATEGORIASACTIVOS';
    ELSIF v_count_relaciones > v_plan.RELACIONES THEN
        RETURN 'INSUFICIENTE: RELACIONES';
    ELSE
        RETURN 'SUFICIENTE';
    END IF;
END;
/
