-- USUARIO ESTÁNDAR
CREATE OR REPLACE VIEW V_USUARIO AS 
SELECT * FROM USUARIO WHERE USUARIO.NOMBREUSUARIO=USER
WITH CHECK OPTION;

GRANT SELECT, UPDATE(NOMBRECOMPLETO, AVATAR, EMAIL, TELEFONO) ON V_USUARIO
TO R_USER_STD;

CREATE PUBLIC SYNONYM MI_USUARIO FOR V_USUARIO;

-- 4. Creamos el sinónimo público
CREATE OR REPLACE PUBLIC SYNONYM MIS_PRODUCTOS FOR V_PRODUCTO_STD;

CREATE OR REPLACE VIEW V_ACTIVO_STD AS 
SELECT ID, NOMBRE, TAMANIO, TIPO, URL
FROM ACTIVO
WHERE ACTIVO.CUENTA_ID IN (SELECT CUENTA_ID FROM V_USUARIO)
WITH CHECK OPTION;

GRANT SELECT, INSERT ON V_ACTIVO_STD TO R_USER_STD;

CREATE PUBLIC SYNONYM MIS_ACTIVOS FOR V_ACTIVO_STD;

-- 1. Primero recreamos la vista sin el WITH CHECK OPTION
CREATE OR REPLACE VIEW V_PRODUCTO_STD AS 
SELECT GTIN, SKU, NOMBRE, MINIATURA, TEXTOCORTO, CREADO, MODIFICADO 
FROM PRODUCTO
WHERE PRODUCTO.CUENTA_ID IN (SELECT CUENTA_ID FROM V_USUARIO);

-- 2. Creamos triggers INSTEAD OF para manejar las operaciones DML
CREATE OR REPLACE TRIGGER trg_v_producto_std_insert
INSTEAD OF INSERT ON V_PRODUCTO_STD
FOR EACH ROW
DECLARE
  v_cuenta_id NUMBER;
BEGIN
  -- Obtenemos la cuenta_id del usuario actual
  SELECT CUENTA_ID INTO v_cuenta_id FROM V_USUARIO WHERE ROWNUM = 1;
  
  -- Insertamos en la tabla base con la cuenta_id del usuario actual
  INSERT INTO PRODUCTO (GTIN, SKU, NOMBRE, MINIATURA, TEXTOCORTO, CREADO, MODIFICADO, CUENTA_ID)
  VALUES (:NEW.GTIN, :NEW.SKU, :NEW.NOMBRE, :NEW.MINIATURA, :NEW.TEXTOCORTO, 
          :NEW.CREADO, :NEW.MODIFICADO, v_cuenta_id);
END;
/

CREATE OR REPLACE TRIGGER trg_v_producto_std_update
INSTEAD OF UPDATE ON V_PRODUCTO_STD
FOR EACH ROW
BEGIN
  UPDATE PRODUCTO
  SET GTIN = :NEW.GTIN,
      SKU = :NEW.SKU,
      NOMBRE = :NEW.NOMBRE,
      MINIATURA = :NEW.MINIATURA,
      TEXTOCORTO = :NEW.TEXTOCORTO,
      MODIFICADO = :NEW.MODIFICADO
  WHERE GTIN = :OLD.GTIN
  AND CUENTA_ID IN (SELECT CUENTA_ID FROM V_USUARIO);
END;
/

CREATE OR REPLACE TRIGGER trg_v_producto_std_delete
INSTEAD OF DELETE ON V_PRODUCTO_STD
FOR EACH ROW
BEGIN
  DELETE FROM PRODUCTO
  WHERE GTIN = :OLD.GTIN
  AND CUENTA_ID IN (SELECT CUENTA_ID FROM V_USUARIO);
END;
/

-- 3. Asignamos todos los permisos CRUD al rol
GRANT SELECT, INSERT, UPDATE, DELETE ON V_PRODUCTO_STD TO R_USER_STD;
