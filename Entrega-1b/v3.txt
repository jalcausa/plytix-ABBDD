// Para administrador del sistema tenemos al usuario PLYTIX, propietario de todas las tablas
-- USUARIO ESTÁNDAR
--Acceso a USUARIO (Usando una vista)
CREATE OR REPLACE VIEW V_USUARIO AS 
SELECT * FROM USUARIO WHERE UPPER(USUARIO.NOMBREUSUARIO)=USER
WITH CHECK OPTION;

CREATE PUBLIC SYNONYM MI_USUARIO FOR V_USUARIO;

GRANT SELECT, UPDATE(NOMBRECOMPLETO, AVATAR, EMAIL, TELEFONO) ON V_USUARIO
TO R_USER_STD;

--Acceso a Productos (usando una vista y triggers en el ejercicio 1 para el CRUD)
-- Hasta que no se crean los triggers no se puede hacer el grant al role
CREATE OR REPLACE VIEW V_PRODUCTO_STD AS 
SELECT GTIN, SKU, NOMBRE, MINIATURA, TEXTOCORTO, CREADO, MODIFICADO 
FROM PRODUCTO
WHERE PRODUCTO.CUENTA_ID IN (SELECT CUENTA_ID FROM V_USUARIO);

CREATE OR REPLACE PUBLIC SYNONYM MIS_PRODUCTOS FOR V_PRODUCTO_STD;

--Acceso a Activo (usando una vista y triggers en el ejercicio 1 para el CRUD)
CREATE OR REPLACE VIEW V_ACTIVO_STD AS 
SELECT ID, NOMBRE, TAMANIO, TIPO, URL
FROM ACTIVO
WHERE ACTIVO.CUENTA_ID IN (SELECT CUENTA_ID FROM V_USUARIO)
WITH CHECK OPTION;

CREATE PUBLIC SYNONYM MIS_ACTIVOS FOR V_ACTIVO_STD;

-- Usando políticas de seguridad:
-- Función de política
CREATE OR REPLACE FUNCTION SOLO_USER_ATRIB(
    p_schema IN VARCHAR2,
    p_object IN VARCHAR2
) RETURN VARCHAR2 AS
BEGIN
    RETURN 'CUENTA_ID IN (SELECT CUENTA_ID FROM V_USUARIO)';
END SOLO_USER_ATRIB;

// Hacer trigger para el insert, con VPD no se puede

CREATE OR REPLACE TRIGGER verificar_cuenta_id
BEFORE INSERT ON atributo
FOR EACH ROW
DECLARE
    v_count NUMBER;
BEGIN
    -- Verificar que el CUENTA_ID del registro a insertar existe en V_USUARIO
    SELECT COUNT(*)
    INTO v_count
    FROM V_USUARIO
    WHERE CUENTA_ID = :NEW.CUENTA_ID;

    -- Si no existe, lanzamos una excepción
    IF v_count = 0 THEN
        RAISE_APPLICATION_ERROR(-20001, 'El CUENTA_ID no es válido o no pertenece al usuario.');
    END IF;
END verificar_cuenta_id;
/

-- EJERCICIO 1a
-- Creamos triggers INSTEAD OF para manejar las operaciones DML sobre PRODUCTO
CREATE OR REPLACE TRIGGER trg_v_producto_std_insert
INSTEAD OF INSERT ON V_PRODUCTO_STD
FOR EACH ROW
DECLARE
  v_cuenta_id NUMBER;
BEGIN
  -- Obtenemos la cuenta_id del usuario actual
  SELECT CUENTA_ID INTO v_cuenta_id FROM V_USUARIO WHERE ROWNUM = 1;
  
  -- Insertamos en la tabla base con la cuenta_id del usuario actual
  INSERT INTO PRODUCTO (GTIN, SKU, NOMBRE, MINIATURA, TEXTOCORTO, CREADO, MODIFICADO, CUENTA_ID)
  VALUES (:NEW.GTIN, :NEW.SKU, :NEW.NOMBRE, :NEW.MINIATURA, :NEW.TEXTOCORTO, 
          :NEW.CREADO, :NEW.MODIFICADO, v_cuenta_id);
END;
/

CREATE OR REPLACE TRIGGER trg_v_producto_std_update
INSTEAD OF UPDATE ON V_PRODUCTO_STD
FOR EACH ROW
BEGIN
  UPDATE PRODUCTO
  SET GTIN = :NEW.GTIN,
      SKU = :NEW.SKU,
      NOMBRE = :NEW.NOMBRE,
      MINIATURA = :NEW.MINIATURA,
      TEXTOCORTO = :NEW.TEXTOCORTO,
      MODIFICADO = :NEW.MODIFICADO
  WHERE GTIN = :OLD.GTIN
  AND CUENTA_ID IN (SELECT CUENTA_ID FROM V_USUARIO);
END;
/

CREATE OR REPLACE TRIGGER trg_v_producto_std_delete
INSTEAD OF DELETE ON V_PRODUCTO_STD
FOR EACH ROW
BEGIN
  DELETE FROM PRODUCTO
  WHERE GTIN = :OLD.GTIN
  AND CUENTA_ID IN (SELECT CUENTA_ID FROM V_USUARIO);
END;
/

-- Ahora sí podemos asignar todos los permisos CRUD al rol
