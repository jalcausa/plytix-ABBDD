// Para administrador del sistema tenemos al usuario PLYTIX, propietario de todas las tablas
-- COMO SYSTEM:
CREATE ROLE R_USER_STD;
CREATE ROLE R_GESTOR_CUENTA;
CREATE ROLE R_PLANIFICADOR_SERVICIOS;

-- USUARIO ESTÁNDAR
CREATE OR REPLACE VIEW V_USUARIO AS 
SELECT * FROM USUARIO WHERE USUARIO.NOMBREUSUARIO=USER
WITH CHECK OPTION;

GRANT SELECT, UPDATE(NOMBRECOMPLETO, AVATAR, EMAIL, TELEFONO) ON V_USUARIO
TO R_USER_STD;

CREATE PUBLIC SYNONYM MI_USUARIO FOR V_USUARIO;

CREATE OR REPLACE VIEW V_PRODUCTO_STD AS 
SELECT GTIN, SKU, NOMBRE, MINIATURA, TEXTOCORTO, CREADO, MODIFICADO 
FROM PRODUCTO
WHERE PRODUCTO.CUENTA_ID IN (SELECT CUENTA_ID FROM V_USUARIO)
WITH CHECK OPTION;

GRANT SELECT, INSERT ON V_PRODUCTO_STD TO R_USER_STD;

CREATE PUBLIC SYNONYM MIS_PRODUCTOS FOR V_PRODUCTO_STD;

CREATE OR REPLACE VIEW V_ACTIVO_STD AS 
SELECT ID, NOMBRE, TAMANIO, TIPO, URL
FROM ACTIVO
WHERE ACTIVO.CUENTA_ID IN (SELECT CUENTA_ID FROM V_USUARIO)
WITH CHECK OPTION;

GRANT SELECT, INSERT ON V_ACTIVO_STD TO R_USER_STD;

CREATE PUBLIC SYNONYM MIS_ACTIVOS FOR V_ACTIVO_STD;


CREATE OR REPLACE VIEW V_ATRIBUTO_STD AS 
SELECT ID, NOMBRE, TIPO, CREADO
FROM ATRIBUTO
WHERE ATRIBUTO.CUENTA_ID IN (SELECT CUENTA_ID FROM V_USUARIO)
WITH CHECK OPTION;

GRANT SELECT, INSERT ON V_ATRIBUTO_STD TO R_USER_STD;

CREATE PUBLIC SYNONYM MIS_ATRIBUTOS FOR V_ATRIBUTO_STD;

-- GESTOR DE CUENTAS
GRANT SELECT, INSERT, UPDATE, DELETE ON CUENTA TO R_GESTOR_CUENTA;

CREATE OR REPLACE VIEW V_USUARIO_GESTOR AS
SELECT ID, NOMBREUSUARIO, NOMBRECOMPLETO, AVATAR
FROM USUARIO;

GRANT SELECT ON V_USUARIO_GESTOR TO R_GESTOR_CUENTA;

CREATE PUBLIC SYNONYM USUARIO FOR USUARIO;

-- PLANIFICADOR DE SERVICIOS
GRANT SELECT, INSERT, UPDATE, DELETE ON PLAN TO R_PLANIFICADOR_SERVICIOS;
