// Hacer trigger para el insert, con VPD no se puede

CREATE OR REPLACE TRIGGER verificar_cuenta_id
BEFORE INSERT ON atributo
FOR EACH ROW
DECLARE
    v_count NUMBER;
BEGIN
    -- Verificar que el CUENTA_ID del registro a insertar existe en V_USUARIO
    SELECT COUNT(*)
    INTO v_count
    FROM V_USUARIO
    WHERE CUENTA_ID = :NEW.CUENTA_ID;

    -- Si no existe, lanzamos una excepción
    IF v_count = 0 THEN
        RAISE_APPLICATION_ERROR(-20001, 'El CUENTA_ID no es válido o no pertenece al usuario.');
    END IF;
END verificar_cuenta_id;

-- Creamos triggers INSTEAD OF para manejar las operaciones DML sobre PRODUCTO
CREATE OR REPLACE TRIGGER trg_v_producto_std_insert
INSTEAD OF INSERT ON V_PRODUCTO_STD
FOR EACH ROW
DECLARE
  v_cuenta_id NUMBER;
BEGIN
  -- Obtenemos la cuenta_id del usuario actual
  SELECT CUENTA_ID INTO v_cuenta_id FROM V_USUARIO WHERE ROWNUM = 1;
  
  -- Insertamos en la tabla base con la cuenta_id del usuario actual
  INSERT INTO PRODUCTO (GTIN, SKU, NOMBRE, MINIATURA, TEXTOCORTO, CREADO, MODIFICADO, CUENTA_ID)
  VALUES (:NEW.GTIN, :NEW.SKU, :NEW.NOMBRE, :NEW.MINIATURA, :NEW.TEXTOCORTO, 
          :NEW.CREADO, :NEW.MODIFICADO, v_cuenta_id);
END;
/

CREATE OR REPLACE TRIGGER trg_v_producto_std_update
INSTEAD OF UPDATE ON V_PRODUCTO_STD
FOR EACH ROW
BEGIN
  UPDATE PRODUCTO
  SET GTIN = :NEW.GTIN,
      SKU = :NEW.SKU,
      NOMBRE = :NEW.NOMBRE,
      MINIATURA = :NEW.MINIATURA,
      TEXTOCORTO = :NEW.TEXTOCORTO,
      MODIFICADO = :NEW.MODIFICADO
  WHERE GTIN = :OLD.GTIN
  AND CUENTA_ID IN (SELECT CUENTA_ID FROM V_USUARIO);
END;
/

CREATE OR REPLACE TRIGGER trg_v_producto_std_delete
INSTEAD OF DELETE ON V_PRODUCTO_STD
FOR EACH ROW
BEGIN
  DELETE FROM PRODUCTO
  WHERE GTIN = :OLD.GTIN
  AND CUENTA_ID IN (SELECT CUENTA_ID FROM V_USUARIO);
END;
/

-- Vista para gestionar la relación entre Activos y Categorías de Activos
CREATE OR REPLACE VIEW V_ACTIVO_CATEGORIA_STD AS
SELECT ACA.*
FROM ACTIVO_CATEGORIA_ACTIVO ACA
WHERE ACA.ACTIVO_CUENTA_ID IN (SELECT CUENTA_ID FROM V_USUARIO)
  AND ACA.CAT_ACT_CUENTA_ID IN (SELECT CUENTA_ID FROM V_USUARIO)
WITH CHECK OPTION;

-- Crear sinónimo público
CREATE PUBLIC SYNONYM MIS_ACTIVOS_CATEGORIAS FOR V_ACTIVO_CATEGORIA_STD;

-- TRIGGERS para CRUD en la vista ACTIVO_CATEGORIA_STD
-- Trigger para INSERT en V_ACTIVO_CATEGORIA_STD
CREATE OR REPLACE TRIGGER trg_v_activo_cat_std_insert
INSTEAD OF INSERT ON V_ACTIVO_CATEGORIA_STD
FOR EACH ROW
DECLARE
    v_cuenta_id NUMBER;
    v_activo_existe NUMBER;
    v_cat_existe NUMBER;
BEGIN
    -- Obtener CUENTA_ID del usuario actual
    SELECT CUENTA_ID INTO v_cuenta_id 
    FROM V_USUARIO 
    WHERE ROWNUM = 1;
    
    -- Verificar que el activo pertenece a la cuenta del usuario
    SELECT COUNT(*) INTO v_activo_existe
    FROM ACTIVO
    WHERE ID = :NEW.ACTIVO_ID
    AND CUENTA_ID = v_cuenta_id;
    
    -- Verificar que la categoría pertenece a la cuenta del usuario
    SELECT COUNT(*) INTO v_cat_existe
    FROM CATEGORIA_ACTIVO
    WHERE ID = :NEW.CAT_ACT_ID
    AND CUENTA_ID = v_cuenta_id;
    
    IF v_activo_existe = 0 THEN
        RAISE_APPLICATION_ERROR(-20012, 'El activo no pertenece a su cuenta');
    END IF;
    
    IF v_cat_existe = 0 THEN
        RAISE_APPLICATION_ERROR(-20013, 'La categoría no pertenece a su cuenta');
    END IF;
    
    -- Insertar la relación
    INSERT INTO ACTIVO_CATEGORIA_ACTIVO (
        ACTIVO_ID, ACTIVO_CUENTA_ID, 
        CAT_ACT_ID, CAT_ACT_CUENTA_ID
    ) VALUES (
        :NEW.ACTIVO_ID, v_cuenta_id, 
        :NEW.CAT_ACT_ID, v_cuenta_id
    );
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        RAISE_APPLICATION_ERROR(-20011, 'Usuario actual no tiene una cuenta asociada');
END;
/

-- Trigger para UPDATE en V_ACTIVO_CATEGORIA_STD
CREATE OR REPLACE TRIGGER trg_v_activo_cat_std_update
INSTEAD OF UPDATE ON V_ACTIVO_CATEGORIA_STD
FOR EACH ROW
DECLARE
    v_cuenta_id NUMBER;
    v_activo_existe NUMBER;
    v_cat_existe NUMBER;
BEGIN
    -- Obtener CUENTA_ID del usuario actual
    SELECT CUENTA_ID INTO v_cuenta_id 
    FROM V_USUARIO 
    WHERE ROWNUM = 1;
    
    -- Verificar que el nuevo activo pertenece a la cuenta del usuario
    SELECT COUNT(*) INTO v_activo_existe
    FROM ACTIVO
    WHERE ID = :NEW.ACTIVO_ID
    AND CUENTA_ID = v_cuenta_id;
    
    -- Verificar que la nueva categoría pertenece a la cuenta del usuario
    SELECT COUNT(*) INTO v_cat_existe
    FROM CATEGORIA_ACTIVO
    WHERE ID = :NEW.CAT_ACT_ID
    AND CUENTA_ID = v_cuenta_id;
    
    IF v_activo_existe = 0 THEN
        RAISE_APPLICATION_ERROR(-20012, 'El nuevo activo no pertenece a su cuenta');
    END IF;
    
    IF v_cat_existe = 0 THEN
        RAISE_APPLICATION_ERROR(-20013, 'La nueva categoría no pertenece a su cuenta');
    END IF;
    
    -- Actualizar la relación
    UPDATE ACTIVO_CATEGORIA_ACTIVO
    SET ACTIVO_ID = :NEW.ACTIVO_ID,
        ACTIVO_CUENTA_ID = v_cuenta_id,
        CAT_ACT_ID = :NEW.CAT_ACT_ID,
        CAT_ACT_CUENTA_ID = v_cuenta_id
    WHERE ACTIVO_ID = :OLD.ACTIVO_ID
    AND ACTIVO_CUENTA_ID = :OLD.ACTIVO_CUENTA_ID
    AND CAT_ACT_ID = :OLD.CAT_ACT_ID
    AND CAT_ACT_CUENTA_ID = :OLD.CAT_ACT_CUENTA_ID
    AND ACTIVO_CUENTA_ID = v_cuenta_id;
    
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        RAISE_APPLICATION_ERROR(-20011, 'Usuario actual no tiene una cuenta asociada');

-- Modificar el trigger TR_PRODUCTOS:
CREATE OR REPLACE TRIGGER TR_PRODUCTOS
BEFORE INSERT ON PRODUCTO FOR EACH ROW
DECLARE
    v_cuenta_id NUMBER;
BEGIN
    -- Asignación automática de GTIN si es nulo
    IF :new.GTIN IS NULL THEN
        :new.GTIN := SEQ_PRODUCTOS.NEXTVAL;
    END IF;
    
    -- Asignación automática de CUENTA_ID
    IF :new.CUENTA_ID IS NULL THEN
        -- Obtener CUENTA_ID del usuario actual
        BEGIN
            SELECT CUENTA_ID 
            INTO v_cuenta_id 
            FROM USUARIO 
            WHERE UPPER(NOMBREUSUARIO) = USER;
            
            :new.CUENTA_ID := v_cuenta_id;
        EXCEPTION
            WHEN NO_DATA_FOUND THEN
                RAISE_APPLICATION_ERROR(-20010, 'Usuario actual no tiene una cuenta asociada');
        END;
    END IF;
END TR_PRODUCTOS;

-- Trigger para INSERT en V_PRODUCTO_CATEGORIA_STD
CREATE OR REPLACE TRIGGER trg_v_prod_cat_std_insert
INSTEAD OF INSERT ON V_PRODUCTO_CATEGORIA_STD
FOR EACH ROW
DECLARE
    v_cuenta_id NUMBER;
    v_prod_existe NUMBER;
    v_cat_existe NUMBER;
    v_ya_tiene_cat NUMBER;
BEGIN
    -- Obtener CUENTA_ID del usuario actual
    SELECT CUENTA_ID INTO v_cuenta_id 
    FROM V_USUARIO 
    WHERE ROWNUM = 1;
    
    -- Verificar que el producto pertenece a la cuenta del usuario
    SELECT COUNT(*) INTO v_prod_existe
    FROM PRODUCTO
    WHERE GTIN = :NEW.PRODUCTO_GTIN
    AND CUENTA_ID = v_cuenta_id;
    
    -- Verificar que la categoría pertenece a la cuenta del usuario
    SELECT COUNT(*) INTO v_cat_existe
    FROM CATEGORIA
    WHERE ID = :NEW.CATEGORIA_ID
    AND CUENTA_ID = v_cuenta_id;
    
    -- Verificar si el producto ya tiene una categoría asignada
    SELECT COUNT(*) INTO v_ya_tiene_cat
    FROM PRODUCTO_CATEGORIA
    WHERE PRODUCTO_GTIN = :NEW.PRODUCTO_GTIN
    AND PRODUCTO_CUENTA_ID = v_cuenta_id;
    
    IF v_prod_existe = 0 THEN
        RAISE_APPLICATION_ERROR(-20022, 'El producto no pertenece a su cuenta');
    END IF;
    
    IF v_cat_existe = 0 THEN
        RAISE_APPLICATION_ERROR(-20023, 'La categoría no pertenece a su cuenta');
    END IF;
    
    IF v_ya_tiene_cat > 0 THEN
        RAISE_APPLICATION_ERROR(-20024, 'El producto ya tiene una categoría asignada. Debe eliminarla primero.');
    END IF;
    
    -- Insertar la relación
    INSERT INTO PRODUCTO_CATEGORIA (
        PRODUCTO_GTIN, PRODUCTO_CUENTA_ID, 
        CATEGORIA_ID, CATEGORIA_CUENTA_ID
    ) VALUES (
        :NEW.PRODUCTO_GTIN, v_cuenta_id, 
        :NEW.CATEGORIA_ID, v_cuenta_id
    );
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        RAISE_APPLICATION_ERROR(-20021, 'Usuario actual no tiene una cuenta asociada');
END;

-- Vista para relaciones entre productos del usuario actual
CREATE OR REPLACE VIEW V_RELACIONADO_STD AS
SELECT R.NOMBRE, R.SENTIDO, 
       R.PRODUCTO_GTIN, R.PRODUCTO_GTIN1
FROM RELACIONADO R
WHERE R.PRODUCTO_CUENTA_ID IN (SELECT CUENTA_ID FROM V_USUARIO)
  AND R.PRODUCTO_CUENTA_ID1 IN (SELECT CUENTA_ID FROM V_USUARIO)
WITH CHECK OPTION;

-- Crear sinónimo público
CREATE PUBLIC SYNONYM MIS_PRODUCTOS_RELACIONADOS FOR V_RELACIONADO_STD;


-- Trigger para INSERT en V_RELACIONADO_STD
CREATE OR REPLACE TRIGGER trg_v_relacionado_std_insert
INSTEAD OF INSERT ON V_RELACIONADO_STD
FOR EACH ROW
DECLARE
    v_cuenta_id NUMBER;
    v_prod1_existe NUMBER;
    v_prod2_existe NUMBER;
BEGIN
    -- Obtener CUENTA_ID del usuario actual
    SELECT CUENTA_ID INTO v_cuenta_id 
    FROM V_USUARIO 
    WHERE ROWNUM = 1;
    
    -- Verificar que el primer producto pertenece a la cuenta del usuario
    SELECT COUNT(*) INTO v_prod1_existe
    FROM PRODUCTO
    WHERE GTIN = :NEW.PRODUCTO_GTIN
    AND CUENTA_ID = v_cuenta_id;
    
    -- Verificar que el segundo producto pertenece a la cuenta del usuario
    SELECT COUNT(*) INTO v_prod2_existe
    FROM PRODUCTO
    WHERE GTIN = :NEW.PRODUCTO_GTIN1
    AND CUENTA_ID = v_cuenta_id;
    
    IF v_prod1_existe = 0 THEN
        RAISE_APPLICATION_ERROR(-20031, 'El primer producto no pertenece a su cuenta');
    END IF;
    
    IF v_prod2_existe = 0 THEN
        RAISE_APPLICATION_ERROR(-20032, 'El segundo producto no pertenece a su cuenta');
    END IF;
    
    -- Evitar relacionar un producto consigo mismo
    IF :NEW.PRODUCTO_GTIN = :NEW.PRODUCTO_GTIN1 THEN
        RAISE_APPLICATION_ERROR(-20033, 'No se puede relacionar un producto consigo mismo');
    END IF;
    
    -- Insertar la relación
    INSERT INTO RELACIONADO (
        NOMBRE, SENTIDO,
        PRODUCTO_GTIN, PRODUCTO_CUENTA_ID, 
        PRODUCTO_GTIN1, PRODUCTO_CUENTA_ID1
    ) VALUES (
        :NEW.NOMBRE, :NEW.SENTIDO,
        :NEW.PRODUCTO_GTIN, v_cuenta_id, 
        :NEW.PRODUCTO_GTIN1, v_cuenta_id
    );
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        RAISE_APPLICATION_ERROR(-20030, 'Usuario actual no tiene una cuenta asociada');
END;
/

-- Trigger para UPDATE en V_RELACIONADO_STD
CREATE OR REPLACE TRIGGER trg_v_relacionado_std_update
INSTEAD OF UPDATE ON V_RELACIONADO_STD
FOR EACH ROW
DECLARE
    v_cuenta_id NUMBER;
    v_prod1_existe NUMBER;
    v_prod2_existe NUMBER;
BEGIN
    -- Obtener CUENTA_ID del usuario actual
    SELECT CUENTA_ID INTO v_cuenta_id 
    FROM V_USUARIO 
    WHERE ROWNUM = 1;
    
    -- Si están cambiando los productos relacionados, verificar que pertenecen a la cuenta
    IF :NEW.PRODUCTO_GTIN != :OLD.PRODUCTO_GTIN OR :NEW.PRODUCTO_GTIN1 != :OLD.PRODUCTO_GTIN1 THEN
        -- Verificar que el primer producto pertenece a la cuenta del usuario
        SELECT COUNT(*) INTO v_prod1_existe
        FROM PRODUCTO
        WHERE GTIN = :NEW.PRODUCTO_GTIN
        AND CUENTA_ID = v_cuenta_id;
        
        -- Verificar que el segundo producto pertenece a la cuenta del usuario
        SELECT COUNT(*) INTO v_prod2_existe
        FROM PRODUCTO
        WHERE GTIN = :NEW.PRODUCTO_GTIN1
        AND CUENTA_ID = v_cuenta_id;
        
        IF v_prod1_existe = 0 THEN
            RAISE_APPLICATION_ERROR(-20031, 'El primer producto no pertenece a su cuenta');
        END IF;
        
        IF v_prod2_existe = 0 THEN
            RAISE_APPLICATION_ERROR(-20032, 'El segundo producto no pertenece a su cuenta');
        END IF;
        
        -- Evitar relacionar un producto consigo mismo
        IF :NEW.PRODUCTO_GTIN = :NEW.PRODUCTO_GTIN1 THEN
            RAISE_APPLICATION_ERROR(-20033, 'No se puede relacionar un producto consigo mismo');
        END IF;
    END IF;
    
    -- Actualizar la relación
    UPDATE RELACIONADO
    SET NOMBRE = :NEW.NOMBRE,
        SENTIDO = :NEW.SENTIDO,
        PRODUCTO_GTIN = :NEW.PRODUCTO_GTIN,
        PRODUCTO_GTIN1 = :NEW.PRODUCTO_GTIN1
    WHERE PRODUCTO_GTIN = :OLD.PRODUCTO_GTIN
      AND PRODUCTO_GTIN1 = :OLD.PRODUCTO_GTIN1
      AND PRODUCTO_CUENTA_ID = v_cuenta_id
      AND PRODUCTO_CUENTA_ID1 = v_cuenta_id;
    
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        RAISE_APPLICATION_ERROR(-20030, 'Usuario actual no tiene una cuenta asociada');
END;
/

-- Trigger para DELETE en V_RELACIONADO_STD
CREATE OR REPLACE TRIGGER trg_v_relacionado_std_delete

WITH CHECK OPTION;

    v_atrib_existe NUMBER;

-- Para la tabla USUARIO creamos una vista limitando la info a la que puede acceder:

CREATE OR REPLACE VIEW V_USUARIO_GESTOR AS
SELECT ID, NOMBREUSUARIO, NOMBRECOMPLETO, AVATAR, CUENTA_ID
FROM USUARIO;

-- Crear sinónimo público
CREATE PUBLIC SYNONYM USUARIOS_GESTION FOR V_USUARIO_GESTOR;

-- Trigger para operaciones INSERT en la vista V_USUARIO_GESTOR
CREATE OR REPLACE TRIGGER trg_v_usuario_gestor_insert
INSTEAD OF INSERT ON V_USUARIO_GESTOR
FOR EACH ROW
BEGIN
    -- Insertar en la tabla base
    INSERT INTO USUARIO (
        ID, NOMBREUSUARIO, NOMBRECOMPLETO, AVATAR, CUENTA_ID
    ) VALUES (
        :NEW.ID, :NEW.NOMBREUSUARIO, :NEW.NOMBRECOMPLETO, 
        :NEW.AVATAR, :NEW.CUENTA_ID
    );
END;
/

-- Trigger para operaciones UPDATE en la vista V_USUARIO_GESTOR
CREATE OR REPLACE TRIGGER trg_v_usuario_gestor_update
INSTEAD OF UPDATE ON V_USUARIO_GESTOR
FOR EACH ROW
BEGIN
    -- Actualizar en la tabla base
    UPDATE USUARIO
    SET NOMBREUSUARIO = :NEW.NOMBREUSUARIO,
        NOMBRECOMPLETO = :NEW.NOMBRECOMPLETO,
        AVATAR = :NEW.AVATAR,
        CUENTA_ID = :NEW.CUENTA_ID
    WHERE ID = :OLD.ID;
END;
/

-- Trigger para operaciones DELETE en la vista V_USUARIO_GESTOR
CREATE OR REPLACE TRIGGER trg_v_usuario_gestor_delete
INSTEAD OF DELETE ON V_USUARIO_GESTOR
FOR EACH ROW
BEGIN
    -- Verificar si el usuario a eliminar es propietario de una cuenta
    DECLARE
        v_es_propietario NUMBER := 0;
    BEGIN
        SELECT COUNT(*) INTO v_es_propietario
        FROM CUENTA
        WHERE USUARIO_ID = :OLD.ID;
        
        IF v_es_propietario > 0 THEN
            RAISE_APPLICATION_ERROR(-20103, 'No se puede eliminar el usuario porque es propietario de una cuenta');
        END IF;
    END;
    
    -- Si no es propietario de ninguna cuenta, permitir la eliminación
    DELETE FROM USUARIO
    WHERE ID = :OLD.ID;
END;
/

-- Asignar permisos de solo lectura sobre la vista de usuarios
GRANT SELECT ON V_USUARIO_GESTOR TO R_GESTOR_CUENTA;
